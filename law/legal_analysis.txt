    "text = \"The court ruled in favor of the plaintiff due to breach of contract.\"\n",

    "print(\"Tokens:\", tokens)\n",
    "\n",
    "# Sentiment analysis\n",
    "sia = SentimentIntensityAnalyzer()\n",
    "sentiment = sia.polarity_scores(text)\n",
