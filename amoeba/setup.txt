#!/bin/bash
# AMOEBA 2048AES SDK Setup Script
# Description: Installs dependencies and sets up the environment for AMOEBA 2048AES SDK

echo "Setting up AMOEBA 2048AES SDK environment..."

# Install system dependencies
sudo apt update
sudo apt install -y python3 python3-pip nvidia-driver-535 nvidia-cuda-toolkit

# Install Python dependencies
pip install --quiet qiskit==1.0.0 torch==2.0.1 pydantic

# Install OCaml and Ortac for formal verification
sudo apt install -y opam
opam init --disable-sandboxing
opam install -y ortac

# Install Docker for containerized execution
curl -fsSL https://get.docker.com -o get-docker.sh
sh get-docker.sh
sudo apt-get install -y nvidia-container-toolkit
sudo systemctl restart docker

# Create project directory
mkdir -p ./amoeba2048_project
cp amoeba_2048_sdk.py amoeba_maml_template.maml.md chimera_spec.mli ./amoeba2048_project/

echo "AMOEBA 2048AES SDK setup complete."
