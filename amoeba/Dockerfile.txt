# AMOEBA 2048AES Orchestrator Dockerfile
# Description: Dockerfile for building a containerized AMOEBA 2048AES orchestrator service, including FastAPI and Dropbox integration.

FROM nvidia/cuda:12.2.0-devel-ubuntu22.04

# Install system dependencies
RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip \
    git \
    && rm -rf /var/lib/apt/lists/*

# Install Python dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Install OCaml and Ortac for verification
RUN apt-get update && apt-get install -y opam && \
    opam init --disable-sandboxing && \
    opam install -y ortac

# Copy project files
WORKDIR /app
COPY amoeba_2048_sdk.py .
COPY security_manager.py .
COPY dropbox_integration.py .
COPY orchestration_dunes.py .
COPY chimera_spec.mli .
COPY dropbox_config.yml .

# Expose port for FastAPI
EXPOSE 8000

# Run orchestrator
CMD ["uvicorn", "orchestration_dunes:app", "--host", "0.0.0.0", "--port", "8000"]
